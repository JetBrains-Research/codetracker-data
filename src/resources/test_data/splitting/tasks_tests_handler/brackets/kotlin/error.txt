package kotlin
import kotlin.math.roundToInt
fun main() {
    val chars = readLine()!!.toCharArray().toMutableList()
    fun p(i: Int, c: Char) : String {
    // hello
        returnwhen {
            i < (chars.size/2.0).roundToInt() - 1 -> "$c("
            i > chars.size/2.0 -> ")$c"
            else -> "$c"
        }
    }
    println(chars.foldIndexed("") {i, str, c -> "$str${p(i, c)}"})
}
