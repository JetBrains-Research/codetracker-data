package java_tasks;

import java.util.Scanner;

public class Zero {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        boolean hasZero = false;
        for (int i = 0; i < n; i++) {
            int next = sc.nextInt();
            if (next == 0) {
                hasZero = true;
            }
        }
        if (hasZero) {
            System.out.print("NO");
        } else
            System.out.print("YES");
    }
}
